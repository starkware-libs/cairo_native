1//! # Struct type
2//!
3//! A struct is just a fixed collection of values that may have different types, which are known at
4//! compile-time. Its fields are properly aligned and respect the declaration's field ordering.
5//!
6//! For example, the following struct would have a layout as described in the table below:
7//!
8//! ```cairo
9//! struct MyStruct {
10//!     U8: u8,
11//!     U16: u16,
12//!     U32: u32,
13//!     U64: u64,
14//!     Felt: Felt,
15//! }
16//! ```
17//!
18//! | Index | Type   | ABI (in Rust types) | Alignment | Size |
19//! | ----- | ------ | ------------------- | --------- | ---- |
20//! |   0   | `i8`   | `u8`                |         1 |    1 |
21//! |  N/A  | N/A    | `[u8; 1]`           |         1 |    1 |
22//! |   1   | `i16`  | `u16`               |         2 |    2 |
23//! |  N/A  | N/A    | `[u8; 2]`           |         1 |    2 |
24//! |   2   | `i32`  | `u32`               |         4 |    4 |
25//! |  N/A  | N/A    | `[u8; 4]`           |         1 |    4 |
26//! |   3   | `i64`  | `u64`               |         8 |    8 |
27//! |   4   | `i252` | `[u64; 4]`          |         8 |    8 |
28//!
29//! As inferred in the table above, the struct will have 8-byte alignment and a size of 30 bytes.
30//! Since this way of generating structs is equivalent to the one used in C and C++, the same
31//! effects apply. For example, if we invert the order of the fields the ABI will change but we
32//! won't waste a single byte in padding; unless we're creating an array, in which case we'd waste
33//! only a single byte per element.
34
35use super::WithSelf;
36use crate::{error::Result, metadata::MetadataStorage, utils::ProgramRegistryExt};

38    extensions::{
39        core::{CoreLibfunc, CoreType},
40        structure::StructConcreteType,
41    },
42    program_registry::ProgramRegistry,
43};
44use melior::{

50/// Build the MLIR type.
51///
52/// Check out [the module](self) for more info.
53pub fn build<'ctx>(
54    context: &'ctx Context,
55    module: &Module<'ctx>,
56    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
57    metadata: &mut MetadataStorage,
58    info: WithSelf<StructConcreteType>,
59) -> Result<Type<'ctx>> {
60    let members = info
61        .members
62        .iter()
63        .map(|member| registry.build_type(context, module, metadata, member))
64        .collect::<Result<Vec<_>>>()?;
65    Ok(llvm::r#type::r#struct(context, &members, false))
66}
