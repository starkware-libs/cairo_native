24/// Build the MLIR type.
25///
26/// Check out [the module](self) for more info.
27pub fn build<'ctx>(
28    context: &'ctx Context,
29    module: &Module<'ctx>,
30    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
31    metadata: &mut MetadataStorage,
32    selector: WithSelf<CircuitTypeConcrete>,
33) -> Result<Type<'ctx>> {
34    match &*selector {
35        CircuitTypeConcrete::CircuitModulus(_) => Ok(IntegerType::new(context, 384).into()),
36        CircuitTypeConcrete::U96Guarantee(_) => Ok(IntegerType::new(context, 96).into()),
37        CircuitTypeConcrete::CircuitInputAccumulator(_) => Ok(llvm::r#type::r#struct(
38            context,
39            &[
40                IntegerType::new(context, 64).into(),
41                llvm::r#type::pointer(context, 0),
42            ],
43            false,
44        )),
45        // Arena-allocated `*u384`.
46        CircuitTypeConcrete::CircuitData(_) => Ok(llvm::r#type::pointer(context, 0)),
47        // Arena-allocated `{ *u384, u384struct }`.
48        CircuitTypeConcrete::CircuitOutputs(_) => Ok(llvm::r#type::r#struct(
49            context,
50            &[
51                llvm::r#type::pointer(context, 0),
52                build_u384_struct_type(context),
53            ],
54            false,
55        )),
56        CircuitTypeConcrete::U96LimbsLessThanGuarantee(info) => {
57            build_u96_limbs_less_than_guarantee(
58                context,
59                module,
60                registry,
61                metadata,
62                WithSelf::new(selector.self_ty(), info),
63            )
64        }
65        // builtins
66        CircuitTypeConcrete::AddMod(_) | CircuitTypeConcrete::MulMod(_) => {
67            Ok(IntegerType::new(context, 64).into())
68        }
69        // noops
70        CircuitTypeConcrete::CircuitDescriptor(_)
71        | CircuitTypeConcrete::CircuitFailureGuarantee(_)
72        | CircuitTypeConcrete::CircuitPartialOutputs(_) => {
73            Ok(llvm::r#type::array(IntegerType::new(context, 8).into(), 0))
74        }
75        // phantoms
76        CircuitTypeConcrete::Circuit(_)
77        | CircuitTypeConcrete::AddModGate(_)
78        | CircuitTypeConcrete::SubModGate(_)
79        | CircuitTypeConcrete::MulModGate(_)
80        | CircuitTypeConcrete::InverseGate(_)
81        | CircuitTypeConcrete::CircuitInput(_) => {
82            Err(SierraAssertError::BadTypeInit(selector.self_ty.clone()))?
83        }
84    }
85}

87pub fn build_u96_limbs_less_than_guarantee<'ctx>(
88    context: &'ctx Context,
89    _module: &Module<'ctx>,
90    _registry: &ProgramRegistry<CoreType, CoreLibfunc>,
91    _metadata: &mut MetadataStorage,
92    info: WithSelf<ConcreteU96LimbsLessThanGuarantee>,
93) -> Result<Type<'ctx>> {
94    let limbs = info.inner.limb_count;
95
96    let u96_type = IntegerType::new(context, 96).into();
97    let limb_struct_type = llvm::r#type::r#struct(context, &vec![u96_type; limbs], false);
98
99    Ok(llvm::r#type::r#struct(
100        context,
101        &[limb_struct_type, limb_struct_type],
102        false,
103    ))
104}

106pub const fn is_complex(info: &CircuitTypeConcrete) -> bool {
107    match *info {
108        CircuitTypeConcrete::AddMod(_)
109        | CircuitTypeConcrete::MulMod(_)
110        | CircuitTypeConcrete::AddModGate(_)
111        | CircuitTypeConcrete::SubModGate(_)
112        | CircuitTypeConcrete::MulModGate(_)
113        | CircuitTypeConcrete::U96Guarantee(_)
114        | CircuitTypeConcrete::InverseGate(_)
115        | CircuitTypeConcrete::U96LimbsLessThanGuarantee(_)
116        | CircuitTypeConcrete::CircuitModulus(_)
117        | CircuitTypeConcrete::CircuitInput(_)
118        | CircuitTypeConcrete::Circuit(_)
119        | CircuitTypeConcrete::CircuitDescriptor(_)
120        | CircuitTypeConcrete::CircuitFailureGuarantee(_) => false,
121
122        CircuitTypeConcrete::CircuitInputAccumulator(_)
123        | CircuitTypeConcrete::CircuitPartialOutputs(_)
124        | CircuitTypeConcrete::CircuitData(_)
125        | CircuitTypeConcrete::CircuitOutputs(_) => true,
126    }
127}

152pub fn layout(
153    _registry: &ProgramRegistry<CoreType, CoreLibfunc>,
154    info: &CircuitTypeConcrete,
155) -> Result<Layout> {
156    match info {
157        CircuitTypeConcrete::AddMod(_) | CircuitTypeConcrete::MulMod(_) => {
158            Ok(get_integer_layout(64))
159        }
160        CircuitTypeConcrete::CircuitModulus(_) => Ok(get_integer_layout(384)),
161        CircuitTypeConcrete::U96Guarantee(_) => Ok(get_integer_layout(96)),
162
163        CircuitTypeConcrete::AddModGate(_)
164        | CircuitTypeConcrete::SubModGate(_)
165        | CircuitTypeConcrete::MulModGate(_)
166        | CircuitTypeConcrete::CircuitInput(_)
167        | CircuitTypeConcrete::InverseGate(_)
168        | CircuitTypeConcrete::U96LimbsLessThanGuarantee(_)
169        | CircuitTypeConcrete::Circuit(_)
170        | CircuitTypeConcrete::CircuitDescriptor(_)
171        | CircuitTypeConcrete::CircuitFailureGuarantee(_) => Ok(Layout::new::<()>()),
172
173        CircuitTypeConcrete::CircuitData(_) => Ok(Layout::new::<*mut ()>()),
174        CircuitTypeConcrete::CircuitOutputs(_) => {
175            let u384_struct_layout = layout_repeat(&get_integer_layout(96), 4)?.0;
176            let pointer_layout = Layout::new::<*mut ()>();
177
178            Ok(pointer_layout.extend(u384_struct_layout)?.0)
179        }
180        CircuitTypeConcrete::CircuitInputAccumulator(_) => {
181            let integer_layout = get_integer_layout(64);
182            let pointer_layout = Layout::new::<*mut ()>();
183
184            Ok(integer_layout.extend(pointer_layout)?.0)
185        }
186        CircuitTypeConcrete::CircuitPartialOutputs(_) => Ok(Layout::new::<()>()),
187    }
188}
