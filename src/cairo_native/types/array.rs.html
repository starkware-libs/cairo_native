1//! # Array type
2//!
3//! An array type is a dynamically allocated list of items.
4//!
5//! ## Layout
6//!
7//! Being dynamically allocated, we just need to keep the pointer to the metadata, its length and
8//! its capacity:
9//!
10//! | Index | Type        | Description                   |
11//! | ----- | ----------- | ----------------------------- |
12//! |   0   | `!llvm.ptr` | Pointer to ArrayMetadata[^1]. |
13//! |   1   | `i32`       | Array start offset[^2].       |
14//! |   2   | `i32`       | Array end offset[^2].         |
15//! |   3   | `i32`       | Allocated capacity[^2].       |
16//!
17//! The ArrayMetadata struct contains:
18//!   1. Array max length (`u32`).
19//!   2. Pointer to array data (`*mut u8`).
20//!
21//! [^1]: This pointer is null when the array has not yet been allocated (i.e., initially).
22//! [^2]: Those numbers are number of items, **not bytes**.
23

40/// Build the MLIR type.
41///
42/// Check out [the module](self) for more info.
43pub fn build<'ctx>(
44    context: &'ctx Context,
45    module: &Module<'ctx>,
46    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
47    metadata: &mut MetadataStorage,
48    info: WithSelf<InfoAndTypeConcreteType>,
49) -> Result<Type<'ctx>> {
50    // Ensure the element type is built.
51    registry.build_type(context, module, metadata, &info.ty)?;
52
53    let ptr_ty = llvm::r#type::pointer(context, 0);
54    let len_ty = IntegerType::new(context, 32).into();
55
56    Ok(llvm::r#type::r#struct(
57        context,
58        &[ptr_ty, len_ty, len_ty, len_ty],
59        false,
60    ))
61}

63/// Metadata struct definition for arrays (maxlen, data_ptr).
64#[repr(C)]
65pub struct ArrayMetadata {
66    pub max_len: u32,
67    pub data_ptr: *mut u8,
68}
69
70/// Returns the metadata struct layout size.
71pub fn calc_metadata_size() -> usize {
72    std::mem::size_of::<ArrayMetadata>()
73}
74
75/// Returns the metadata struct alignment.
76pub fn calc_metadata_align() -> usize {
77    std::mem::align_of::<ArrayMetadata>()
78}
79
80/// Get the LLVM struct type for ArrayMetadata: { max_len: i32, data_ptr: ptr }
81///
82/// Field indices:
83/// - 0: max_len (u32)
84/// - 1: data_ptr (*mut u8)
85pub fn get_metadata_llvm_type(context: &Context) -> Type<'_> {
86    llvm::r#type::r#struct(
87        context,
88        &[
89            IntegerType::new(context, 32).into(), // max_len: u32
90            llvm::r#type::pointer(context, 0),    // data_ptr: *mut u8
91        ],
92        false,
93    )
94}
95
96#[repr(i32)]
97enum MetadataField {
98    MaxLen = 0,
99    DataPtr = 1,
100}
101
102/// Returns a pointer to the given `field` in the ArrayMetadata struct.
103fn metadata_field_ptr<'ctx, 'this>(
104    context: &'ctx Context,
105    block: &'this Block<'ctx>,
106    location: Location<'ctx>,
107    metadata_ptr: Value<'ctx, 'this>,
108    field: MetadataField,
109) -> Result<Value<'ctx, 'this>> {
110    Ok(block.gep(
111        context,
112        location,
113        metadata_ptr,
114        &[GepIndex::Const(0), GepIndex::Const(field as i32)],
115        get_metadata_llvm_type(context),
116    )?)
117}

119/// Load `data_ptr` from the ArrayMetadata struct.
120pub fn load_data_ptr<'ctx, 'this>(
121    context: &'ctx Context,
122    block: &'this Block<'ctx>,
123    location: Location<'ctx>,
124    metadata_ptr: Value<'ctx, 'this>,
125) -> Result<Value<'ctx, 'this>> {
126    let field_ptr = metadata_field_ptr(
127        context,
128        block,
129        location,
130        metadata_ptr,
131        MetadataField::DataPtr,
132    )?;
133    Ok(block.load(
134        context,
135        location,
136        field_ptr,
137        llvm::r#type::pointer(context, 0),
138    )?)
139}

141/// Store `max_len` (i32) into the ArrayMetadata struct.
142pub fn store_max_len<'ctx, 'this>(
143    context: &'ctx Context,
144    block: &'this Block<'ctx>,
145    location: Location<'ctx>,
146    metadata_ptr: Value<'ctx, 'this>,
147    max_len: Value<'ctx, 'this>,
148) -> Result<()> {
149    let field_ptr = metadata_field_ptr(
150        context,
151        block,
152        location,
153        metadata_ptr,
154        MetadataField::MaxLen,
155    )?;
156    block.store(context, location, field_ptr, max_len)?;
157    Ok(())
158}
