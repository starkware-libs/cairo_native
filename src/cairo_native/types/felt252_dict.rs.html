1//! # `Felt` dictionary type
2//!
3//! A key value storage for values whose type implement Copy. The key is always a felt.
4//!
5//! This type is represented as a pointer to a tuple of a heap allocated Rust hashmap along with a u64
6//! used to count accesses to the dictionary. The type is interacted through the runtime functions to
7//! insert, get elements and increment the access counter.
8
9use super::WithSelf;
10use crate::{error::Result, metadata::MetadataStorage};
11use cairo_lang_sierra::{
12    extensions::{
13        core::{CoreLibfunc, CoreType},
14        types::InfoAndTypeConcreteType,
15    },
16    program_registry::ProgramRegistry,
17};
18use melior::{
19    dialect::llvm,
20    ir::{Module, Type},
21    Context,
22};

24/// Build the MLIR type.
25///
26/// Dict snapshots are semantically useless in Cairo, so the default dup
27/// (returns the same pointer twice) is correct. The arena owns dict memory and
28/// HashMaps are reclaimed via DICT_REGISTRY during arena reset, so the default
29/// drop (noop) is also correct. No dup/drop overrides are registered.
30///
31/// Check out [the module](self) for more info.
32pub fn build<'ctx>(
33    context: &'ctx Context,
34    _module: &Module<'ctx>,
35    _registry: &ProgramRegistry<CoreType, CoreLibfunc>,
36    _metadata: &mut MetadataStorage,
37    _info: WithSelf<InfoAndTypeConcreteType>,
38) -> Result<Type<'ctx>> {
39    Ok(llvm::r#type::pointer(context, 0))
40}
