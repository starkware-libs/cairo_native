6//! ## ContractAddress

40/// Build the MLIR type.
41///
42/// Check out [the module](self) for more info.
43pub fn build<'ctx>(
44    context: &'ctx Context,
45    module: &Module<'ctx>,
46    registry: &ProgramRegistry<CoreType, CoreLibfunc>,
47    metadata: &mut MetadataStorage,
48    selector: WithSelf<StarknetTypeConcrete>,
49) -> Result<Type<'ctx>> {
50    match &*selector {
51        StarknetTypeConcrete::ClassHash(info) => build_class_hash(
52            context,
53            module,
54            registry,
55            metadata,
56            WithSelf::new(selector.self_ty(), info),
57        ),
58        StarknetTypeConcrete::ContractAddress(info) => build_contract_address(
59            context,
60            module,
61            registry,
62            metadata,
63            WithSelf::new(selector.self_ty(), info),
64        ),
65        StarknetTypeConcrete::StorageBaseAddress(info) => build_storage_base_address(
66            context,
67            module,
68            registry,
69            metadata,
70            WithSelf::new(selector.self_ty(), info),
71        ),
72        StarknetTypeConcrete::StorageAddress(info) => build_storage_address(
73            context,
74            module,
75            registry,
76            metadata,
77            WithSelf::new(selector.self_ty(), info),
78        ),
79        StarknetTypeConcrete::System(info) => build_system(
80            context,
81            module,
82            registry,
83            metadata,
84            WithSelf::new(selector.self_ty(), info),
85        ),
86        StarknetTypeConcrete::Secp256Point(info) => build_secp256_point(
87            context,
88            module,
89            registry,
90            metadata,
91            WithSelf::new(selector.self_ty(), info),
92        ),
93        StarknetTypeConcrete::Sha256StateHandle(info) => build_sha256_state_handle(
94            context,
95            module,
96            registry,
97            metadata,
98            WithSelf::new(selector.self_ty(), info),
99        ),
100        StarknetTypeConcrete::Sha512StateHandle(info) => build_sha512_state_handle(
101            context,
102            module,
103            registry,
104            metadata,
105            WithSelf::new(selector.self_ty(), info),
106        ),
107    }
108}

154pub fn build_system<'ctx>(
155    context: &'ctx Context,
156    _module: &Module<'ctx>,
157    _registry: &ProgramRegistry<CoreType, CoreLibfunc>,
158    _metadata: &mut MetadataStorage,
159    _info: WithSelf<InfoOnlyConcreteType>,
160) -> Result<Type<'ctx>> {
161    Ok(llvm::r#type::pointer(context, 0))
162}

164pub fn build_secp256_point<'ctx>(
165    context: &'ctx Context,
166    _module: &Module<'ctx>,
167    _registry: &ProgramRegistry<CoreType, CoreLibfunc>,
168    _metadata: &mut MetadataStorage,
169    _info: WithSelf<Secp256PointTypeConcrete>,
170) -> Result<Type<'ctx>> {
171    Ok(llvm::r#type::r#struct(
172        context,
173        &[
174            llvm::r#type::r#struct(
175                context,
176                &[
177                    IntegerType::new(context, 128).into(),
178                    IntegerType::new(context, 128).into(),
179                ],
180                false,
181            ),
182            llvm::r#type::r#struct(
183                context,
184                &[
185                    IntegerType::new(context, 128).into(),
186                    IntegerType::new(context, 128).into(),
187                ],
188                false,
189            ),
190            IntegerType::new(context, 1).into(),
191        ],
192        false,
193    ))
194}

196pub fn build_sha256_state_handle<'ctx>(
197    context: &'ctx Context,
198    _module: &Module<'ctx>,
199    _registry: &ProgramRegistry<CoreType, CoreLibfunc>,
200    _metadata: &mut MetadataStorage,
201    _info: WithSelf<InfoOnlyConcreteType>,
202) -> Result<Type<'ctx>> {
203    // Arena-allocated [u32; 8]. No dup/drop overrides needed — bitwise
204    // copy of the pointer is safe and drop is a noop (arena owns memory).
205    Ok(llvm::r#type::pointer(context, 0))
206}
